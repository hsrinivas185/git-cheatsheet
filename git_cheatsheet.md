## to initialize git repository on the local machine
git init

## To know the status of the git repository
git status

## to add untracked files to the staging area
git add -A

## to commit the changes to the branch
git commit -am "add cheatsheet file"

## to get the repository log(history)
git log

## to get the hostory on the repo in graph format
gloga/ git log --oneline --decorate --graph --all

## to checkout to a perticular commit with commit id
git checkout sha1-hash

## to switch branches
git switch branch
